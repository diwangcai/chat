# 服务器修复命令
# 请在阿里云控制台云助手中执行以下命令

# 1. 检查磁盘空间
df -h

# 2. 清理磁盘空间
npm cache clean --force
sudo apt-get clean
sudo apt-get autoremove

# 3. 安装PM2
npm install -g pm2

# 4. 进入项目目录
cd /home/shell

# 5. 清理并重新安装依赖
rm -rf node_modules package-lock.json
npm install --production

# 6. 构建项目
npm run build

# 7. 启动应用
pm2 start npm --name "chat-app" -- start

# 8. 检查状态
pm2 status
netstat -tlnp | grep :3000

# 9. 查看日志
pm2 logs chat-app
